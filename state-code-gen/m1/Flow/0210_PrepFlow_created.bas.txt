Sub PrepFlow_Select(func As String, bFirst As Boolean)
    Select Case func
        Case "S_START"
        	S_START(bFirst)
        Case "S_GETTPL"
        	S_GETTPL(bFirst)
        Case "S_GETPROP"
        	S_GETPROP(bFirst)
        Case "S_STATELIST"
        	S_STATELIST(bFirst)
        Case "S_END"
        	S_END(bFirst)
        
    End Select
End Sub
'
' 開始
Sub S_START(bFirst As Boolean)
    If bFirst = True Then
            SetNextState("S_GETTPL")
    End If
End Sub
' テンプレート取得
Sub S_GETTPL(bFirst As Boolean)
    If bFirst = True Then
            SetNextState("S_GETPROP")
            call tplsrc_get
            call tplfunc_get
    End If
End Sub
' プロパティの取得
Sub S_GETPROP(bFirst As Boolean)
    If bFirst = True Then
            SetNextState("S_STATELIST")
            call props_get
    End If
End Sub
' ステートリスト作成
Sub S_STATELIST(bFirst As Boolean)
    If bFirst = True Then
            SetNextState("S_END")
    End If
End Sub
' 終了
Sub S_END(bFirst As Boolean)
    If bFirst = True Then
            SetNextState("S_END")
    End If
End Sub
